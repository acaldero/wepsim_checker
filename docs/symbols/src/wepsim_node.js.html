 21     /**
 22      * Initialize WepSIM core
 23      */
 24     function wepsim_nodejs_init ( )
 25     {
 26         reset_cfg() ;
 27         stop_drawing() ;
 28 
 29         check_behavior();
 30         compile_behaviors() ;
 31         firedep_to_fireorder(jit_fire_dep) ;
 32         compute_references() ;
 33     }
 34 
 35     /**
 36      * Reset the WepSIM simulation
 37      */
 38     function wepsim_nodejs_reset ( )
 39     {
 40 	var SIMWARE = get_simware() ;
 41         compute_general_behavior("RESET") ;
 42 
 43         if ((typeof segments['.ktext'] != "undefined") && (SIMWARE.labels2["kmain"])){
 44                     set_value(sim_states["REG_PC"], parseInt(SIMWARE.labels2["kmain"]));
 45 	}
 46         else if ((typeof segments['.text'] != "undefined") && (SIMWARE.labels2["main"])){
 47                     set_value(sim_states["REG_PC"], parseInt(SIMWARE.labels2["main"]));
 48 	}
 49 
 50 	if ( (typeof segments['.stack'] != "undefined") &&
 51              (typeof sim_states["BR"][FIRMWARE.stackRegister] != "undefined") )
 52 	{
 53 		set_value(sim_states["BR"][FIRMWARE.stackRegister], parseInt(segments['.stack'].begin));
 54 	}
 55 
 56 	var mode = get_cfg('ws_mode');
 57 	if ('webmips' != mode) {
 58             compute_general_behavior("CLOCK") ;
 59 	}
 60     }

 62     /**
 63      * Compile Firmware
 64      * @param {string} textToMCompile - The firmware to be compile and loaded into memory
 65      */
 66     function wepsim_nodejs_compile_firmware ( textToMCompile )
 67     {
 68 	var ret = new Object() ;
 69 	    ret.msg = "" ;
 70 	    ret.ok  = true ;
 71 
 72 	var preSM = load_firmware(textToMCompile) ;
 73 	if (preSM.error != null)
 74         {
 75             ret.msg = preSM.error ;
 76             ret.ok  = false ;
 77             return ret ;
 78         }
 79 
 80 	wepsim_nodejs_reset() ;
 81         return ret ;
 82     }

 84     /**
 85      * Compile Assembly
 86      * @param {string} textToCompile - The assembly to be compile and loaded into memory
 87      */
 88     function wepsim_nodejs_compile_assembly ( textToCompile )
 89     {
 90 	var ret = new Object() ;
 91 	    ret.msg = "" ;
 92 	    ret.ok  = true ;
 93 
 94         // get SIMWARE.firmware
 95         var SIMWARE = get_simware() ;
 96 	if (SIMWARE.firmware.length == 0)
 97         {
 98             ret.msg = 'WARNING: please load the microcode first.' ;
 99             ret.ok  = false;
100             return ret;
101 	}
102 
103         // compile Assembly and show message
104         var SIMWAREaddon = simlang_compile(textToCompile, SIMWARE);
105         if (SIMWAREaddon.error != null)
106         {
107             ret.msg = SIMWAREaddon.error ;
108             ret.ok  = false;
109             return ret;
110         }
111 
112         // update memory and segments
113         set_simware(SIMWAREaddon) ;
114 	update_memories(SIMWARE) ;
115 	wepsim_nodejs_reset() ;
116         return ret ;
117     }

119     /**
120      * Execute the assembly previously compiled and loaded
121      * @param {integer} ins_limit - The limit of instructions to be executed
122      * @param {integer} clk_limit - The limit of clock cycles per instruction
123      */
124     function wepsim_nodejs_execute ( ins_limit, clk_limit )
125     {
126 	var ret = new Object() ;
127 	    ret.error = false ;
128 	    ret.msg   = "" ;
129 
130         // execute firmware-assembly
131         wepsim_nodejs_init() ;
132 	wepsim_nodejs_reset() ;
133 
134 	var reg_pc        = sim_states["REG_PC"].value ;
135 	var reg_pc_before = sim_states["REG_PC"].value - 4 ;
136 
137 	var code_begin  = 0 ;
138 	if ( (typeof segments['.text'] != "undefined") && (typeof segments['.text'].begin != "undefined") )
139 	      code_begin = parseInt(segments['.text'].begin) ;
140 	var code_end    = 0 ;
141 	if ( (typeof segments['.text'] != "undefined") && (typeof segments['.text'].end   != "undefined") )
142 	      code_end = parseInt(segments['.text'].end) ;
143 
144 	var kcode_begin = 0 ;
145 	if ( (typeof segments['.ktext'] != "undefined") && (typeof segments['.ktext'].begin != "undefined") )
146 	      kcode_begin = parseInt(segments['.ktext'].begin) ;
147 	var kcode_end   = 0 ;
148 	if ( (typeof segments['.ktext'] != "undefined") && (typeof segments['.ktext'].end   != "undefined") )
149 	      kcode_end = parseInt(segments['.ktext'].end) ;
150 
151 	var ins_executed = 0 ; 
152 	while (
153                      (reg_pc != reg_pc_before)  &&
154                   ( ((reg_pc <  code_end) && (reg_pc >=  code_begin)) ||
155                     ((reg_pc < kcode_end) && (reg_pc >= kcode_begin)) )
156               )
157 	{
158 	       ret = execute_microprogram(clk_limit) ;
159                if (false == ret.ok) {
160 		   return ret ;
161 	       }
162 
163 	       ins_executed++ ; 
164                if (ins_executed > ins_limit) 
165 	       {
166 	           ret.error = true ;
167 	           ret.msg   = "more than " + ins_limit + " instructions executed before application ends.";
168 		   return ret ;
169 	       }
170 
171 	       reg_pc_before = reg_pc ;
172 	       reg_pc = sim_states["REG_PC"].value ;
173 	}
174 
175         return ret ;
176     }

178     /**
179      * Check that the current state meets the specifications
180      * @param {object} checklist_ok - Correct state specifications
181      */
182     function wepsim_nodejs_check_results ( checklist_ok )
183     {
184 	var data3_bin   = wepsim_checklist2state(checklist_ok) ;
185 	var obj_current = wepsim_current2state();
186 	var obj_result  = wepsim_check_results(data3_bin, obj_current, true) ;
187 
188 	// console.log(JSON.stringify(obj_result.result, null, 2));
189 
190         return obj_result ;
191     }

193     /**
194      * Check that the current state meets the specifications
195      * @param {object}  checkresults - Results checked
196      * @param {string}  show_format - "HTML" or "txt"
197      * @param {boolean} show_onlyerrors - True if only errors has to been shown
198      */
199     function wepsim_nodejs_show_checkresults ( checkresults, show_format, show_onlyerrors )
200     {
201 	if (show_format.toUpperCase() == "HTML") {
202             return wepsim_checkreport2html(checkresults, show_onlyerrors) ;
203 	}
204 
205         return wepsim_checkreport2txt(checkresults) ;
206     }
